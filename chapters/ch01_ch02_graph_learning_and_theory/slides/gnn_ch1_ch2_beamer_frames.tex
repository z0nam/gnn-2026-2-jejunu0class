\section{Chapter 1: Getting Started with Graph Learning}

\begin{frame}{왜 그래프인가?}
\begin{itemize}
  \item 그래프는 노드(객체)와 엣지(관계)를 함께 모델링
  \begin{itemize}
    \item 노드: 엣지 없는 노드는 전통적인 데이터 표현으로도 가능
    \item 엣지: 반드시 2개 이상의 노드 포인트를 포함하며 관계 정보를 표현
  \end{itemize}
  \item 관계 정보가 핵심인 도메인에서 강력
  \item 예: 소셜/교통/생물/추천 시스템
  \item GNN: 관계 구조를 활용한 딥러닝 모델링 프레임워크
\end{itemize}

\end{frame}

\begin{frame}{Graph Learning의 대표 과제}
\small
\begin{itemize}
  \item \textbf{Node Classification}: 각 노드의 라벨(클래스)을 예측
  \item \textbf{Link Prediction}: 두 노드 사이의 연결 존재 여부를 예측
  \item \textbf{Graph Classification}: 그래프 전체(분자/문서/세션)에 대한 라벨을 예측
  \item \textbf{Graph Generation}: 조건을 만족하는 새로운 그래프 구조를 생성
\end{itemize}
\vspace{0.3em}
\footnotesize
예: 사용자 관심사 분류(Node), 친구 추천(Link), 분자 독성 분류(Graph), 신규 분자 설계(Generation)
\end{frame}

\begin{frame}{Graph Learning이 빠르게 발전한 이유}
\begin{itemize}
  \item 그래프 학습은 다양한 도메인에 적용 가능해 연구/개발 가치가 큼
  \item 최근 발전의 핵심 동력:
  \begin{itemize}
    \item 대규모 데이터셋 가용성 증가
    \item 고성능 컴퓨팅 자원 확장
    \item ML/AI 방법론의 빠른 발전
  \end{itemize}
\end{itemize}
\vspace{0.3em}
\textbf{대표 기술 계열 4가지}
\begin{enumerate}
  \item Graph Signal Processing
  \item Matrix Factorization
  \item Random Walk
  \item Deep Learning
\end{enumerate}
\end{frame}

\begin{frame}{4가지 Graph Learning Technique Family}
\small
\begin{itemize}
  \item \textbf{Graph Signal Processing}: 그래프 푸리에 변환/스펙트럴 분석으로 연결 구조의 내재 특성을 분석
  \item \textbf{Matrix Factorization}: 큰 관계 행렬을 저차원 잠재 표현으로 분해해 패턴을 compact하게 표현
  \item \textbf{Random Walk}: 그래프 위 탐색 경로를 시뮬레이션해 노드 간 관계 정보를 수집하고 학습 데이터 생성
  \item \textbf{Deep Learning}: 다층 신경망으로 그래프를 벡터로 인코딩해 다양한 다운스트림 과제에서 높은 성능 달성
\end{itemize}
\vspace{0.3em}
\footnotesize
실무에서는 이 기법들이 서로 배타적이라기보다, 혼합되어 함께 사용되는 경우가 많다.
\end{frame}

\begin{frame}{왜 GNN인가?}
\begin{itemize}
  \item 기존 ML과의 차이: 기존 ML은 노드 feature에만 집중하는 반면, GNN은 노드 특징과 관계 구조를 함께 활용
  \item 기존 그래프이론과의 차이: 그래프 이론은 주로 그래프의 수학적 성질과 알고리즘에 초점, GNN은 그래프 구조를 활용한 표현 학습과 예측에 초점
  \item 이웃 정보를 반복 집계(message passing)
  \item 노드 특징 + 관계 구조를 함께 반영
  \item 대규모/고복잡도 문제에서 표현학습에 유리
  \item 기존 graph ML은 handcrafed features + shallow models이 많았지만, GNN은 end-to-end 학습 가능한 deep architecture로 발전
  \item GNN은 complex graph data에서 높은 표현력과 일반화 능력을 보여주며, 다양한 도메인에서 state-of-the-art 성능 달성
  \item 하지만 GNN은 충분한 데이터가 필요, 복잡한 문제에는 효과적이나 작은 데이터셋의 간단한 문제는 기존 ML이 더 적합할 수 있음
\end{itemize}

\begin{frame}[fragile]{NetworkX 코드 예시: 그래프 생성과 해석}
\begin{lstlisting}
import networkx as nx
import matplotlib.pyplot as plt

# 무방향 그래프를 하나 만든다.
G = nx.Graph()

# 각 튜플은 두 노드 사이의 연결(edge)을 뜻한다.
G.add_edges_from([
    ("A", "B"), ("A", "C"),
    ("B", "D"), ("B", "E"),
    ("C", "F"), ("C", "G"),
])

# 노드 A의 차수(degree)는 연결된 이웃의 수다.
print(f"deg(A) = {G.degree['A']}")

# 시각화용 배치를 계산한 뒤 그래프를 그린다.
pos = nx.spring_layout(G, seed=0)
plt.axis("off")
nx.draw_networkx(G, pos=pos, node_size=600, font_color="white")
\end{lstlisting}
\end{frame}

\begin{frame}{중요 그래프 유형}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.36\textwidth}
  \scriptsize
\begin{itemize}
  \item Tree / Rooted Tree: 트리는 사이클이 없는 연결 그래프. 루트 트리는 트리의 한 노드를 루트로 지정한 그래프
  \item DAG: 유향+사이클 없는 그래프. 트리보다 일반적이며, 순서/의존성 표현에 유용
  \item Bipartite Graph: 노드 집합이 두 개로 나뉘고, 같은 집합 내에서는 엣지가 없는 그래프. 예: 사용자-상품 네트워크
  \item Complete Graph: 모든 노드가 서로 연결된 그래프. 예: 소셜 네트워크에서 모든 사람이 친구인 경우
\end{itemize}
\vspace{0.4em}
\scriptsize
각 유형은 특정 도메인과 문제 유형에서 자주 등장하며, GNN 모델링 시에도 구조적 제약으로 활용될 수 있음

\begin{frame}{기초 객체와 측도}
\begin{itemize}
  \item Degree: $\deg(v)$, Directed(-:in, +:out): $\deg^-(v),\deg^+(v)$
  \item Path, Cycle, Acyclic
  \item Centrality:
  \begin{itemize}
    \item Degree centrality
    \item Closeness centrality
    \item Betweenness centrality
  \end{itemize}
  \item Density:
\[
\rho_{\text{undir}}=\frac{|E|}{|V|(|V|-1)/2},\qquad
\rho_{\text{dir}}=\frac{|E|}{|V|(|V|-1)}
\]
\end{itemize}
\end{frame}

\begin{frame}[fragile]{IPYNB 예제: Degree 계산}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.58\textwidth}
\begin{lstlisting}
G = nx.Graph()
G.add_edges_from([
    ("A", "B"), ("A", "C"),
    ("B", "D"), ("B", "E"),
    ("C", "F"), ("C", "G"),
])

# 무방향 그래프에서 degree는
# 해당 노드에 연결된 이웃 수다.
print(f"deg(A) = {G.degree['A']}")

DG = nx.DiGraph()
DG.add_edges_from([
    ("A", "B"), ("A", "C"),
    ("B", "D"), ("B", "E"),
    ("C", "F"), ("C", "G"),
])

# 방향 그래프에서는 in/out degree를 구분한다.
print(f"deg^-(A) = {DG.in_degree['A']}")
print(f"deg^+(A) = {DG.out_degree['A']}")
\end{lstlisting}
\end{column}
\begin{column}{0.38\textwidth}
\small
\textbf{실행 결과}
\begin{verbatim}
deg(A) = 2
deg^-(A) = 0
deg^+(A) = 2
\end{verbatim}
\vspace{0.4em}
\begin{itemize}
  \item A는 B, C와 연결되어 degree가 2
  \item 방향 그래프에서는 들어오는 엣지와 나가는 엣지를 분리해 본다
\end{itemize}
\end{column}
\end{columns}

\begin{column}{0.4\textwidth}
\scriptsize
\textbf{핵심 해석}
\begin{itemize}
  \item Degree centrality: 직접 연결이 많은 노드가 큼
  \item Closeness centrality: 다른 노드까지 평균 거리가 짧을수록 큼
  \item Betweenness centrality: 최단경로를 자주 중개하는 노드가 큼
  \item 이 예제에서는 중심부에 있는 A, B, C가 말단 노드보다 더 중요하게 측정된다
\end{itemize}
\end{column}
\end{columns}
\vspace{0.4em}
\footnotesize
노트북 출력에서도 A, B, C의 centrality 값이 D, E, F, G보다 높게 나온다.
\end{frame}

\begin{frame}[fragile]{BFS와 DFS 알고리즘}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.56\textwidth}
\begin{lstlisting}
def bfs(graph, node):
    visited, queue = [node], [node]
    while queue:
        node = queue.pop(0)
        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.append(neighbor)
                queue.append(neighbor)
    return visited

def dfs(visited, graph, node):
    if node not in visited:
        visited.append(node)
        for neighbor in graph[node]:
            visited = dfs(visited, graph, neighbor)
    return visited
\end{lstlisting}
\end{column}
\begin{column}{0.4\textwidth}
\small
\textbf{노트북 실행 결과}
\begin{verbatim}
bfs(G, 'A')
['A', 'B', 'C',
 'D', 'E', 'F', 'G']

dfs([], G, 'A')
['A', 'B', 'D',
 'E', 'C', 'F', 'G']
\end{verbatim}
\vspace{0.3em}
\begin{itemize}
  \item BFS는 가까운 레벨부터 넓게 방문
  \item DFS는 한 경로를 끝까지 따라간 뒤 되돌아온다
\end{itemize}
\end{column}
\end{columns}
